}

  let result: int;
  result = firstNumber * secondNumber;
  return result;
}

function divideTwoNumbers: float(firstNumber:float, secondNumber: float) {
  return (firstNumber / secondNumber);
}

function sumFrom0ToN: int(givenLimit: int) {
  let i, result: int;
  for(i = 0 to givenLimit by 1) {
    print(i);
    result = result + i;
  }
  return result;
}
